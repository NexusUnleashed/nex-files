import data from "./classleads_data.json";

export const skillToClasses = Object.entries(data.classes || {}).reduce(
  (acc, [className, skills]) => {
    (skills || []).forEach((skill) => {
      if (!acc[skill]) {
        acc[skill] = [];
      }
      acc[skill].push(className);
    });
    return acc;
  },
  {}
);

export const allSkills = Array.from(
  new Set([
    ...Object.keys(data.skills || {}),
    ...Object.values(data.classes || {}).flat(),
    ...(data.unmappedSkills || []),
  ])
).sort((a, b) => a.localeCompare(b));

function skillSlug(value) {
  return value
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/(^-|-$)+/g, '');
}

export const toc = allSkills.map((skill) => ({
  value: skill,
  id: skillSlug(skill),
  level: 2,
  children: [],
}));

Drag-and-drop update: replace `static/classleads_data.json` and rebuild. The list below renders from that JSON.

<div>
  {allSkills.map((skill) => {
    const changes = data.skills?.[skill] ?? [];
    const classes = skillToClasses[skill] ?? [];
    const skillId = skillSlug(skill);
    const skillSummary = `${skill} ${classes.join(" ")}`.trim();

    return (
      <details
        key={skill}
        id={skillId}
        data-skill-block
        data-text={skillSummary}
      >
        <summary>
          <strong>{skill}</strong>
          {classes.length ? (
            <span style={{ marginLeft: "0.5rem", fontWeight: 400 }}>
              ({classes.join(", ")})
            </span>
          ) : null}
        </summary>
        {!classes.length ? (
          <p>
            <em>Not currently mapped to a class.</em>
          </p>
        ) : null}
        {changes.length ? (
          <ul>
            {changes.map((chg, index) => (
              <li
                key={index}
                data-change-line
                data-text={`${skill} ${chg.text} ${chg.dateDisplay || ""}`}
              >
                {chg.text} {chg.dateDisplay ? <em>({chg.dateDisplay})</em> : null}
              </li>
            ))}
          </ul>
        ) : (
          <p>No changes recorded.</p>
        )}
      </details>
    );

})}

</div>
